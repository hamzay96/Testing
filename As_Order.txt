#include <iostream>
#include <algorithm>

int main() {
    int arr[] = {100, 60, 800, 6, 10, 3}; % Branch 2
    int n = sizeof(arr) / sizeof(arr[0]);

    std::sort(arr, arr + n);   // Ascending order

    std::cout << "Sorted array: ";
    for (int i = 0; i < n; i++) {
        std::cout << arr[i] << " ";
    }

    return 0;
}

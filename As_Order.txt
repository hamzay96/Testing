#include <iostream>
#include <algorithm>

int main() {
    int arr[] = {5, 2, 5, 6, 10, 3};
    int n = sizeof(arr) / sizeof(arr[0]);

    std::sort(arr, arr + n);   // Ascending order

    std::cout << "Sorted array: ";
    for (int i = 0; i < n; i++) {
        std::cout << arr[i] << " ";
    }

    return 0;
}
